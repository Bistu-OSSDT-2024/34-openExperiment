<script>
    let display = document.getElementById('display');

    function clearDisplay() {
        display.value = '';
    }

    function appendToDisplay(value) {
        display.value += value;
    }

    function calculate() {
        try {
            // 使用eval()函数计算表达式的值
            let result = eval(display.value);
            // 检查结果是否为无穷大（除以0的情况）
            if (isFinite(result)) {
                display.value = result;
            } else {
                display.value = 'Error';
            }
        } catch (e) {
            display.value = 'Error';
        }
    }
</script>
